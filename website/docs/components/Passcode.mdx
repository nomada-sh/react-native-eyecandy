## Import

```jsx
import { Passcode } from '@nomada-sh/react-native-eyecandy';
```

## Example

```SnackPlayer name=RN%20Eyecandy%20Passcode
import React from 'react';
import { SafeAreaView, View } from 'react-native';

import {
  TextInput,
  Passcode
} from '@nomada-sh/react-native-eyecandy';

export default function App() {
  const [value, setValue] = React.useState('');

  return (
    <SafeAreaView
      style={{
        flex: 1,
      }}
    >
      <View
        style={{
          flex: 1,
          margin: 20,
          alignItems: 'center',
        }}
      >
        <TextInput
          marginBottom={20}
          value={value}
          onChangeText={setValue}
        />
        <Passcode
          value={value}
          onChange={setValue}
        />
      </View>
    </SafeAreaView>
  );
}
```

### `KeyComponent`

Component to use as key. You need to pass onPressIn and onPressOut props to a pressable component (or any other component that can be pressed).

**Example:**

```jsx
const SquaredKey = ({
  keyValue,
  isDeleteKey,
  onPressIn,
  onPressOut,
  col,
}: PasscodeKeyProps) => (
  <Pressable
    style={{
      flex: 1,
      borderTopWidth: 1,
      borderRightWidth: col < 3 ? 1 : 0,
      borderColor: 'grey',
      justifyContent: 'center',
      alignItems: 'center',
      paddingVertical: 20,
    }}
    onPressIn={onPressIn}
    onPressOut={onPressOut}
  >
    <Body size={24}>{isDeleteKey ? <Backspace /> : keyValue}</Body>
  </Pressable>
);
```
